def myfunc(int1: int):
    # Initialize count to 1
    count = 1
    # Loop while int1 is not 1
    while int1 != 1:
        # If int1 is even, divide by 2
        if int1 % 2 == 0:
            int1 = int1 // 2
        # If int1 is odd, multiply by 3 and add 1
        else:
            int1 = 3 * int1 + 1
        # Increment count
        count += 1
    # Return count
    return count
